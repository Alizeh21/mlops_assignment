pipeline {
    agent any

    stages {

        stage('Checkout') {
            steps {
                git 'https://github.com/Alizeh21/mlops-kubeflow-assignment.git'
            }
        }

        stage('Install Dependencies') {
            steps {
                sh 'pip install -r requirements.txt'
            }
        }

        stage('Run Training Script') {
            steps {
                sh 'python src/model_training.py'
            }
        }

        stage('Build Docker Image') {
            steps {
                sh 'docker build -t mlops-pipeline .'
            }
        }
    }
}
